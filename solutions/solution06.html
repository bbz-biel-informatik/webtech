var image = document.querySelector('img')
image.addEventListener('click', scale);

var large = false;

function scale() {
  if(large) {
    image.style.width = '';
    large = false;
  }
  else {
    image.style.width = '80%';
    large = true;
  }
}

Aufgabe 3 
var all = document.querySelectorAll("*");

for(var i = 0; i < all.length; i++) {
	all[i].addEventListener('mouseover', changeColor);
    all[i].addEventListener('mouseout', changeColorBack);
}

function changeColor(e) {
  this.style.border = '1px solid red';
  e.stopPropagation();
}

function changeColorBack(e) {
  this.style.border = 'none';
  e.stopPropagation();
}
